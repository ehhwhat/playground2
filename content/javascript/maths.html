Number to square

Takes a number as an input and multiples it by itself.

function numberToSquareFunc(input){
    result = input * input;
    document.getElementById("numberToSquareResult").innerHTML += "<p class='display-3 text-success'>" +input+ "<sup>2</sup> = " +result+ "</p>";
    return result;
}

document.getElementById('numberToSquareSubmit').onclick = function() {
    var value = document.getElementById('numberToSquare').value;
    numberToSquareFunc(value);
};
